add_library(util
    arcball_camera.cpp
    imgui_impl_sdl.cpp
    imgui_impl_opengl3.cpp
    shader.cpp
    util.cpp
    material.cpp
    mesh.cpp
    scene.cpp
    buffer_view.cpp
    gltf_types.cpp
    flatten_gltf.cpp
    gl_core_4_5.c)

    CXX_STANDARD 14
    CXX_STANDARD_REQUIRED ON)

    $<BUILD_INTERFACE:${CMAKE_CURRENT_LIST_DIR}>
    $<BUILD_INTERFACE:${SDL2_INCLUDE_DIR}>
    $<BUILD_INTERFACE:${OPENGL_INCLUDE_DIR}>
    $<BUILD_INTERFACE:${GLM_INCLUDE_DIRS}>)

    imgui
    ${SDL2_LIBRARY}
    ${OPENGL_LIBRARIES})
